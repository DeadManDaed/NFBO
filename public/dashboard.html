      // Variables globales pour stocker les informations utilisateur
      let currentUser = {
      username: '',
      role: '',
      userId: null
      };
      // Initialisation au chargement de la page
      document.addEventListener('DOMContentLoaded', async () => {
      // Récupérer les informations utilisateur depuis sessionStorage
      const userInfo = sessionStorage.getItem('userInfo');
      if (!userInfo) {
      // Si pas d'info utilisateur, rediriger vers la page de connexion
      window.location.href = '/index.html';
      return;
      }
      currentUser = JSON.parse(userInfo);
      // Afficher les informations utilisateur dans le header
      displayUserInfo();
      // Charger le tableau de bord selon le rôle
      await loadDashboard();
      });
      // Affichage des informations utilisateur
      function displayUserInfo() {
      document.getElementById('userName').textContent = currentUser.username;
      document.getElementById('userRole').textContent = getRoleLabel(currentUser.role);
      document.getElementById('userAvatar').textContent = currentUser.username.charAt(0).toUpperCase();
      // Message de bienvenue personnalisé
      document.getElementById('welcomeTitle').textContent = `Bienvenue, ${currentUser.username} !`;
      document.getElementById('welcomeText').textContent = getWelcomeMessage(currentUser.role);
      }
      // Obtenir le label du rôle en français
      function getRoleLabel(role) {
      const roles = {
      'admin': 'Administrateur',
      'caisse': 'Gestionnaire de caisse',
      'stock': 'Gestionnaire de stock'
      };
      return roles[role] || 'Utilisateur';
      }
      // Message de bienvenue selon le rôle
      function getWelcomeMessage(role) {
      const messages = {
      'admin': 'Vous avez accès à toutes les fonctionnalités de gestion de la coopérative.',
      'caisse': 'Gérez les opérations de caisse et suivez les transactions des producteurs.',
      'stock': 'Supervisez les stocks et les mouvements de produits dans les magasins.'
      };
      return messages[role] || 'Bienvenue sur votre tableau de bord.';
      }
      // Chargement du tableau de bord selon le rôle
      async function loadDashboard() {
      try {
      // Charger les statistiques
      await loadStatistics();
      // Charger le contenu spécifique au rôle
      switch(currentUser.role) {
      case 'admin':
      await loadAdminDashboard();
      break;
      case 'caisse':
      await loadCaisseDashboard();
      break;
      case 'stock':
      await loadStockDashboard();
      break;
      default:
      loadDefaultDashboard();
      }
      } catch (error) {
      console.error('Erreur lors du chargement du tableau de bord:', error);
      document.getElementById('roleContent').innerHTML = `
      <div class="content-card">
      <p style="color: #c62828;">Erreur lors du chargement des données. Veuillez rafraîchir la page.</p>
      </div>
      `;
      }
      }
      // Chargement des statistiques générales
      async function loadStatistics() {
      const statsGrid = document.getElementById('statsGrid');
      try {
      // Appels API pour récupérer les statistiques
      const [lots, operations, stock, users] = await Promise.all([
      fetch('/lots').then(r => r.json()),
      fetch('/operations').then(r => r.json()),
      fetch('/stock').then(r => r.json()),
      currentUser.role === 'admin' ? fetch('/users').then(r => r.json()) : Promise.resolve([])
      ]);
      // Calcul des statistiques
      const totalLots = lots.length;
